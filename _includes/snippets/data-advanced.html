private DataFileSystem dataFile; private void Init() { dataFile = new DataFileSystem($"{Interface.Oxide.DataDirectory}\\player_info"); } private PlayerInfo LoadPlayerInfo(ulong playerID) { string path = $"playerInfo_{playerID}"; return dataFile.ReadObject(path); } private void SavePlayerInfo(ulong playerID, PlayerInfo playerInfo) { string path = $"playerInfo_{playerID}"; dataFile.WriteObject(path, playerInfo); }

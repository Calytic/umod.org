[Command("get")] private void GetRequest(IPlayer player, string command, string[] args) { // Set a custom timeout (in milliseconds) var timeout = 200f; // Set some custom request headers (eg. for HTTP Basic Auth) var headers = new Dictionary { { "header", "value" } }; webrequest.EnqueueGet("http://www.google.com/search?q=umod", (code, response) => GetCallback(code, response, player), this, headers, timeout); } private void GetCallback(int code, string response, IPlayer player) { if (response == null || code != 200) { Puts($"Error: {code} - Couldn't get an answer from Google for {player.Name}"); return; } Puts($"Google answered for {player.Name}: {response}"); }
